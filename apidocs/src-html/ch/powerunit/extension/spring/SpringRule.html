020package ch.powerunit.extension.spring;
021
022import org.springframework.context.ApplicationContext;
023
024import ch.powerunit.extension.spring.impl.SpringRuleImpl;
025import ch.powerunit.rules.TestListenerRule;
026
027/**
028 * This is a {@link ch.powerunit.TestRule TestRule} to provide support to use
029 * Spring inside PowerUnit.
030 * 
031 * @author borettim
032 *
033 */
034public interface SpringRule extends TestListenerRule {
035        ApplicationContext getApplicationContext();
036
037        /**
038         * Get a bean from the used ApplicationContext.
039         * 
040         * @param requiredType
041         *            the requiredType.
042         * @return the bean
043         * @see org.springframework.context.ApplicationContext#getBean(Class)
044         * @param <T>
045         *            THe type of the bean.
046         */
047        <T> T getBean(Class<T> requiredType);
048
049        /**
050         * Create a rule to support Spring.
051         * <p>
052         * For instance:
053         * 
054         * <pre>
055         * &#064;Rule
056         * public final SpringRule spring = SpringRule.of(
057         *              AutowireCapableBeanFactory.AUTOWIRE_BY_NAME, &quot;classpath:sample.xml&quot;);
058         * </pre>
059         * 
060         * @param autowireMode
061         *            The autowiring mode (of the test class).
062         * @param location
063         *            the first location for the bean context.
064         * @param nextLocation
065         *            optional additional location for the bean context.
066         * @return the Rule.
067         * @see org.springframework.beans.factory.config.AutowireCapableBeanFactory#AUTOWIRE_BY_NAME
068         * @see org.springframework.beans.factory.config.AutowireCapableBeanFactory#AUTOWIRE_BY_TYPE
069         */
070        static SpringRule of(int autowireMode, String location,
071                        String... nextLocation) {
072                String[] tmp = new String[nextLocation.length + 1];
073                tmp[0] = location;
074                System.arraycopy(nextLocation, 0, tmp, 1, nextLocation.length);
075                return new SpringRuleImpl(tmp, autowireMode);
076        }
077
078}
