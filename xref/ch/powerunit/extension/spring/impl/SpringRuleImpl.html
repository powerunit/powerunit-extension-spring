20  package ch.powerunit.extension.spring.impl;
21  
22  import org.springframework.beans.BeansException;
23  import org.springframework.context.ApplicationContext;
24  import org.springframework.context.support.ClassPathXmlApplicationContext;
25  
26  import ch.powerunit.TestContext;
27  import ch.powerunit.extension.spring.SpringRule;
28  
29  /**
30   * @author borettim
31   *
32   */
33  public final class SpringRuleImpl implements SpringRule {
34  
35  	private ClassPathXmlApplicationContext springContext;
36  
37  	private final String location[];
38  
39  	private final int autowireMode;
40  
41  	/**
42  	 * @param location
43  	 * @param autowireMode
44  	 */
45  	public SpringRuleImpl(String[] location, int autowireMode) {
46  		this.location = location;
47  		this.autowireMode = autowireMode;
48  	}
49  
50  	/*
51  	 * (non-Javadoc)
52  	 * 
53  	 * @see
54  	 * ch.powerunit.rules.TestListenerRule#onStart(ch.powerunit.TestContext)
55  	 */
56  	@Override
57  	public void onStart(TestContext<Object> context) {
58  		springContext = new ClassPathXmlApplicationContext(location);
59  		springContext.refresh();
60  		springContext.getAutowireCapableBeanFactory().autowireBeanProperties(
61  				context.getTestSuiteObject(), autowireMode, true);
62  	}
63  
64  	/*
65  	 * (non-Javadoc)
66  	 * 
67  	 * @see ch.powerunit.rules.TestListenerRule#onEnd(ch.powerunit.TestContext)
68  	 */
69  	@Override
70  	public void onEnd(TestContext<Object> context) {
71  		if (springContext != null && springContext.isActive()) {
72  			springContext.close();
73  		}
74  	}
75  
76  	/*
77  	 * (non-Javadoc)
78  	 * 
79  	 * @see ch.powerunit.extension.spring.SpringRule#getApplicationContext()
80  	 */
81  	@Override
82  	public ApplicationContext getApplicationContext() {
83  		return springContext;
84  	}
85  }
